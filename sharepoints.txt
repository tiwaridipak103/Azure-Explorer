import os
import shutil
from datetime import datetime, timedelta

# Path to the directory containing the folders
directory = '/path/to/your/directory'

# Get the current time
now = datetime.now()

# Define the time limit (2 days ago)
time_limit = now - timedelta(days=2)

# Iterate through all the folders in the directory
for folder_name in os.listdir(directory):
    folder_path = os.path.join(directory, folder_name)
    
    # Check if it's a directory
    if os.path.isdir(folder_path):
        # Get the folder's modification time
        folder_mod_time = datetime.fromtimestamp(os.path.getmtime(folder_path))
        
        # If the folder is older than 2 days, delete it
        if folder_mod_time < time_limit:
            shutil.rmtree(folder_path)  # Use rmtree to delete the folder and its contents
            print(f"Deleted folder: {folder_name}")
